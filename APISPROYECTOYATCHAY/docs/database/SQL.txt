CREATE DATABASE SistemaQuizYatchay;
GO

USE SistemaQuizYatchay;
GO

-- TABLAS DE AUTENTICACIÓN
CREATE TABLE Rol (
    id_rol INT IDENTITY(1,1) PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL UNIQUE
);
GO

INSERT INTO Rol (nombre) VALUES ('ESTUDIANTE'), ('DOCENTE'), ('ADMINISTRADOR');
GO

CREATE TABLE Usuario (
    id_usuario INT IDENTITY(1,1) PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    apellido VARCHAR(100) NOT NULL,
    correo VARCHAR(150) NOT NULL UNIQUE,
    contrasena_hash VARCHAR(255) NOT NULL,
    contrasena_literal VARCHAR(255) NULL,
    dni VARCHAR(20) NOT NULL UNIQUE,
    id_rol INT NOT NULL,
    fecha_registro DATETIME NOT NULL DEFAULT GETDATE(),
    fecha_actualizacion DATETIME NULL,
    CONSTRAINT FK_Usuario_Rol FOREIGN KEY (id_rol) REFERENCES Rol(id_rol)
);
GO

-- TABLAS DEL SIMULADOR YACHAY-TECH
CREATE TABLE SimulationContent (
    id_content INT IDENTITY(1,1) PRIMARY KEY,
    id_content_externo VARCHAR(100) NOT NULL UNIQUE,
    fase VARCHAR(50) NOT NULL,
    categoria VARCHAR(100) NULL,
    tipo VARCHAR(50) NOT NULL,
    titulo NVARCHAR(255) NOT NULL,
    intro NVARCHAR(MAX) NULL,
    pregunta NVARCHAR(MAX) NOT NULL,
    requerida BIT NOT NULL DEFAULT 1,
    shuffle_opciones BIT NOT NULL DEFAULT 1,
    orden INT NOT NULL,
    creado_at DATETIME NOT NULL DEFAULT GETDATE(),
    actualizado_at DATETIME NULL
);
GO

CREATE TABLE SimulationOption (
    id_option INT IDENTITY(1,1) PRIMARY KEY,
    id_option_externo VARCHAR(120) NOT NULL UNIQUE,
    id_content INT NOT NULL,
    texto NVARCHAR(MAX) NOT NULL,
    puntaje DECIMAL(5,2) NOT NULL DEFAULT 0,
    nivel VARCHAR(20) NULL,
    feedback NVARCHAR(MAX) NULL,
    resultado NVARCHAR(MAX) NULL,
    insight NVARCHAR(MAX) NULL,
    siguiente_pregunta_id VARCHAR(100) NULL,
    orden INT NOT NULL DEFAULT 0,
    creado_at DATETIME NOT NULL DEFAULT GETDATE(),
    actualizado_at DATETIME NULL,
    CONSTRAINT FK_Option_Content FOREIGN KEY (id_content) REFERENCES SimulationContent(id_content)
);
GO

CREATE TABLE SimulationSession (
    id_session INT IDENTITY(1,1) PRIMARY KEY,
    id_usuario INT NOT NULL,
    fase_actual VARCHAR(50) NOT NULL DEFAULT 'inicio',
    estado VARCHAR(20) NOT NULL DEFAULT 'EN_PROGRESO',
    puntaje_total DECIMAL(5,2) NOT NULL DEFAULT 0,
    iniciado_at DATETIME NOT NULL DEFAULT GETDATE(),
    finalizado_at DATETIME NULL,
    CONSTRAINT FK_SimulationSession_Usuario FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario)
);
GO

CREATE TABLE Decision (
    id_decision INT IDENTITY(1,1) PRIMARY KEY,
    id_session INT NOT NULL,
    id_content INT NOT NULL,
    id_option INT NOT NULL,
    puntaje_obtenido DECIMAL(5,2) NOT NULL,
    decidido_at DATETIME NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_Decision_Session FOREIGN KEY (id_session) REFERENCES SimulationSession(id_session),
    CONSTRAINT FK_Decision_Content FOREIGN KEY (id_content) REFERENCES SimulationContent(id_content),
    CONSTRAINT FK_Decision_Option FOREIGN KEY (id_option) REFERENCES SimulationOption(id_option),
    UNIQUE(id_session, id_content)
);
GO

-- ÍNDICES
CREATE INDEX IX_SimulationSession_Usuario ON SimulationSession(id_usuario);
CREATE INDEX IX_Decision_Session ON Decision(id_session);
CREATE INDEX IX_SimulationContent_Fase ON SimulationContent(fase);
GO
