CREATE TABLE Usuario (
    id_usuario          INT IDENTITY(1,1) PRIMARY KEY,
    nombre              VARCHAR(100)    NOT NULL,
    apellido            VARCHAR(100)    NOT NULL,
    correo              VARCHAR(150)    NOT NULL UNIQUE,
    contrasena_hash     VARCHAR(255)    NOT NULL,
    contrasena_literal  VARCHAR(255)    NULL,
    dni                 VARCHAR(20)     NOT NULL UNIQUE,
    id_rol              INT             NOT NULL,
    fecha_registro      DATETIME        NOT NULL DEFAULT GETDATE(),
    fecha_actualizacion DATETIME        NULL,
    CONSTRAINT FK_Usuario_Rol FOREIGN KEY (id_rol) REFERENCES Rol(id_rol)
);
GO

CREATE PROCEDURE sp_Usuario_Insertar
    @Nombre NVARCHAR(100),
    @Apellido NVARCHAR(100),
    @Correo NVARCHAR(150),
    @ContrasenaHash NVARCHAR(255),
    @ContrasenalLiteral NVARCHAR(255),
    @DNI NVARCHAR(20),
    @IdRol INT
AS
BEGIN
    IF @Correo NOT LIKE 'i[0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]@cibertec.edu.pe'
    BEGIN
        SELECT 0 AS Exito, 'El correo debe tener formato: i#########@cibertec.edu.pe' AS Mensaje;
        RETURN;
    END
    
    -- Validar formato del DNI (8 dígitos)
    IF @DNI NOT LIKE '[0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]'
    BEGIN
        SELECT 0 AS Exito, 'El DNI debe tener exactamente 8 dígitos' AS Mensaje;
        RETURN;
    END
    
    IF NOT EXISTS (SELECT 1 FROM Rol WHERE id_rol = @IdRol)
    BEGIN
        SELECT 0 AS Exito, 'Rol inválido' AS Mensaje;
        RETURN;
    END
    
    IF EXISTS (SELECT 1 FROM Usuario WHERE correo = @Correo)
    BEGIN
        SELECT 0 AS Exito, 'Correo ya registrado' AS Mensaje;
        RETURN;
    END
    
    IF EXISTS (SELECT 1 FROM Usuario WHERE dni = @DNI)
    BEGIN
        SELECT 0 AS Exito, 'DNI ya registrado' AS Mensaje;
        RETURN;
    END
    
    INSERT INTO Usuario (nombre, apellido, correo, contrasena_hash, contrasena_literal, dni, id_rol)
    VALUES (@Nombre, @Apellido, @Correo, @ContrasenaHash, @ContrasenalLiteral, @DNI, @IdRol);
    
    SELECT 1 AS Exito, 'Usuario registrado' AS Mensaje;
END
GO

CREATE PROCEDURE sp_Usuario_Login
    @Correo NVARCHAR(150),
    @ContrasenaHash NVARCHAR(255)
AS
BEGIN
    SELECT 
        u.id_usuario AS IdUsuario,      
        u.nombre AS Nombre,       
        u.id_rol AS IdRol,              
        r.nombre AS NombreRol       
    FROM Usuario u
    INNER JOIN Rol r ON u.id_rol = r.id_rol
    WHERE u.correo = @Correo AND u.contrasena_hash = @ContrasenaHash;
END
GO

-- TABLAS DEL SIMULADOR
CREATE TABLE SimulationContent (
    id_content INT IDENTITY(1,1) PRIMARY KEY,
    fase INT NOT NULL,
    tipo VARCHAR(50) NOT NULL,
    titulo NVARCHAR(255) NOT NULL,
    opciones NVARCHAR(MAX) NOT NULL, -- JSON con las opciones
    feedback NVARCHAR(MAX) NOT NULL, -- JSON con feedback
    orden INT NOT NULL,
    creado_at DATETIME NOT NULL DEFAULT GETDATE(),
    actualizado_at DATETIME NULL
);
GO

CREATE TABLE SimulationSession (
    id_session INT IDENTITY(1,1) PRIMARY KEY,
    id_usuario INT NOT NULL,
    fase_actual INT NOT NULL DEFAULT 1,
    estado VARCHAR(20) NOT NULL DEFAULT 'EN_PROGRESO', -- EN_PROGRESO, COMPLETADO
    puntaje_total INT NOT NULL DEFAULT 0,
    iniciado_at DATETIME NOT NULL DEFAULT GETDATE(),
    finalizado_at DATETIME NULL,
    CONSTRAINT FK_SimulationSession_Usuario FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario)
);
GO

CREATE TABLE Decision (
    id_decision INT IDENTITY(1,1) PRIMARY KEY,
    id_session INT NOT NULL,
    id_content INT NOT NULL,
    fase INT NOT NULL,
    opcion_elegida INT NOT NULL,
    puntaje_obtenido INT NOT NULL,
    decidido_at DATETIME NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_Decision_SimulationSession FOREIGN KEY (id_session) REFERENCES SimulationSession(id_session),
    CONSTRAINT FK_Decision_SimulationContent FOREIGN KEY (id_content) REFERENCES SimulationContent(id_content),
    -- UNIQUE para inmutabilidad: una decisión por contenido por sesión
    UNIQUE(id_session, id_content)
);
GO

-- ÍNDICES para optimizar búsquedas
CREATE INDEX IX_SimulationSession_Usuario ON SimulationSession(id_usuario);
CREATE INDEX IX_Decision_Session ON Decision(id_session);
CREATE INDEX IX_Decision_Session_Content ON Decision(id_session, id_content);
CREATE INDEX IX_SimulationContent_Fase ON SimulationContent(fase);
GO
